	\begin{frame}{Mathematical Content}{Theorems and Proofs}
	    \begin{definition}[odd integer]
	    	An integer $z \in \mathbb{Z}$ is said to be odd if it is not divisible by two, i.e. there exist no $k \in \mathbb Z$ s.t. $z = 2k$.
	    \end{definition}

	    \begin{theorem}[Multiplication of Odd Integers Yields Odd Integer]
	    	Let $a, b \in \mathbb Z$ be two non-null odd integers. Then $a \cdot b$ is an odd integer.
	    \end{theorem}

	    \begin{proof}
	    	Let $k, l \in \mathbb{Z} \backslash \{0\}, a = 2k+1, b = 2l+1$. Then
		    $$
		    a \cdot b = (2k+1)(2l+1) = 4kl + 2k + 2l + 1 = 2(2kl + k + l) + 1
		    $$
		    which is odd.
	    \end{proof}
	\end{frame} 


	\begin{frame}{Algorithmic Content}{Bubble Sort}
	    Bubble Sort is an algorithm to sort an array of real numbers.
	    \begin{algorithm}[H]
		\caption{BubbleSort($A$)}
		\begin{algorithmic}[1]
		  \State $n \gets \text{length}(A)$
		  \For{$i \gets 1$ \dots $n-1$}
		    \For{$j \gets 1$ \dots $n-i$}
		      \If{$A[j+1] < A[j]$}
		        \State \textbf{swap} $A[j]$ \textbf{and} $A[j+1]$
		      \EndIf
		    \EndFor
		  \EndFor
		  \State \Return $A$
		\end{algorithmic}
		\end{algorithm}
	\end{frame}
